145  /*
146  operation may be one of:
147  (1) predefined OCL operation
148  (2) attribute operation on object type (no arguments)
149  (3) "isQuery" operation on object type (possibly with arguments)
150  (4) navigation operation on object type (with possibly with rolename or qualifiers)
151  (5) shorthand for collect
152  (6) set operation on single object resulting from
153  navigation over associations with multiplicity zero or one
154  (p. 7-13 of OMG UML 1.3)
155  (7) variable
156 
157  possible combinations of syntax:
158  source expression:
159  none or
160  s: simple value
161  o: object
162  c: collection
163 
164  s.op 110 (1)
165  s.op() 111 (1)
166  s->op 120 (1) with warning
167  s->op() 121 (1) with warning
168 
169  o.op 210 (2,4,1)
170  o.op[nav] 1210 (4)
171  o.op() 211 (3,1)
172  o->op 220 (6) if s has object type and results from navigating 0..1 end
173  o->op() 221 (6) to allow uniform syntax of operation calls
174 
175  c.op 410 (5) with implicit (2,4,1)
176  c.op() 411 (5) with implicit (3,1)
177  c->op 420 (1)
178  c->op() 421 (1)
179 
180  op 000 (2,4,7)
181  op() 001 (1,3)
182  */

184  public Expression gen(Context ctx) throws SemanticException {
185  Expression res = null;
186  Expression srcExpr = null;
187  String opname = fOp.getText();
188 
189  if (fSrcExpr != null ) {
190  srcExpr = fSrcExpr.gen(ctx);
191  res = gen1(ctx, srcExpr);
192  } else {
193  // if no source expression is given, it is either a
194  // variable or a reference to expression determined by
195  // context. In the latter case we use the context as
196  // source expression and proceed as if it has been given
197  // explicitly.
198  if (! fHasParentheses ) {
199  // variable?
200  // (7) check for variable
201  Type type = ctx.varTable().lookup(opname);
202  if (type != null )
203  res = new ExpVariable(opname, type);
204  }
205 
206  // do we have a context expression that is implicitly
207  // assumed to be the source expression?
208  if (res == null ) {
209  ExprContext ec = ctx.exprContext();
210  if (! ec.isEmpty() ) {
211  // construct source expression
212  ExprContext.Entry e = ec.peek();
213  srcExpr = new ExpVariable(e.fName, e.fType);
214  if (e.fType.isKindOfCollection(VoidHandling.EXCLUDE_VOID) )
215  fFollowsArrow = true;
216  res = gen1(ctx, srcExpr);
217  } else
218  throw new SemanticException(fOp, "Undefined " +
219  ( fHasParentheses ? "operation" : "variable" ) +
220  " `" + opname + "'.");
221  }
222  }
223 
224  if (isPre() ) {
225  if (! (ctx.insidePostCondition() || ctx.isInsideTestCase() ))
226  throw new SemanticException(fOp,
227  "Modifier @pre is only allowed in postconditions.");
228  res.setIsPre();
229  }
230 
231  if (opname.equals("oclIsNew") ) {
232  if (! ctx.insidePostCondition() )
233  throw new SemanticException(fOp,
234  "Operation oclIsNew is only allowed in postconditions.");
235  }
236 
237  return res;
238  }

241  private Expression gen1(Context ctx, Expression srcExpr)
242  throws SemanticException
243  {
244  Expression res = null;
245  String opname = fOp.getText();
246  Type srcType = srcExpr.type();
247 
248  // handles cases like
249  // !! set C.f().a := 2 (= set attribute a of object C.f() to 2)
250  // where f() is an operation without return value.
251  if (srcType == null) {
252  throw new SemanticException(
253  fOp,
254  "'" +
255  opname +
256  "' cannot be applied to '" +
257  srcExpr +
258  "', since the latter doesn't evaluate to something with a type.");
259  }
260 
261  // generate argument expressions
262  fArgExprs = new Expression[fArgs.size() + 1];
263  fArgExprs[0] = srcExpr;
264  int i = 1;
265 
266  for (ASTExpression astExpr : fArgs) {
267  fArgExprs[i++] = astExpr.gen(ctx);
268  }
269 
270  // flags for various cases
271  final int SRC_SIMPLE_TYPE = 0x0100;
272  final int SRC_OBJECT_TYPE = 0x0200;
273  final int SRC_COLLECTION_TYPE = 0x0400;
274  final int SRC_TUPLE_TYPE = 0x0800;
275 
276  final int DOT = 0x0010;
277  final int ARROW = 0x0020;
278 
279  final int NO_EXPLICIT_ROLENAME = 0x0000;
280  final int EXPLICIT_ROLENAME = 0x1000;
281 
282  final int NO_PARENTHESES = 0x0000;
283  final int PARENTHESES = 0x0001;
284 
285  int opcase;
286  if (srcType.isKindOfClassifier(VoidHandling.EXCLUDE_VOID) )
287  opcase = SRC_OBJECT_TYPE;
288  else if (srcType.isKindOfCollection(VoidHandling.EXCLUDE_VOID) )
289  opcase = SRC_COLLECTION_TYPE;
290  else if (srcType.isKindOfTupleType(VoidHandling.EXCLUDE_VOID) )
291  opcase = SRC_TUPLE_TYPE;
292  else
293  opcase = SRC_SIMPLE_TYPE;
294 
295  opcase += fFollowsArrow ? ARROW : DOT;
296  opcase += fHasParentheses ? PARENTHESES : NO_PARENTHESES;
297  opcase += fExplicitRolenameOrQualifiers.size() > 0 ? EXPLICIT_ROLENAME : NO_EXPLICIT_ROLENAME;
298 
299  switch ( opcase ) {
300  case SRC_SIMPLE_TYPE + DOT + NO_PARENTHESES:
301  case SRC_SIMPLE_TYPE + DOT + PARENTHESES:
302  case SRC_COLLECTION_TYPE + ARROW + PARENTHESES:
303  case SRC_COLLECTION_TYPE + ARROW + NO_PARENTHESES:
304  // (1) predefined OCL operation
305  res = genStdOperation(ctx, fOp, opname, fArgExprs);
306  break;
307 
308  case SRC_SIMPLE_TYPE + ARROW + NO_PARENTHESES:
309  case SRC_SIMPLE_TYPE + ARROW + PARENTHESES:
310  // If source is null literal, convert to Bag{} (OCL 2.3.1 p. 146)
311  if (fArgExprs[0].type().isTypeOfVoidType()) {
312  try {
313  fArgExprs[0] = new ExpBagLiteral(new Expression[0]);
314  } catch (ExpInvalidException e) { }
315  } else {
316  ctx.reportWarning(fOp, "application of `" + opname +
317  "' to a single value should be done with `.' " +
318  "instead of `->'.");
319  }
320 
321  // (1) predefined OCL operation
322  res = genStdOperation(ctx, fOp, opname, fArgExprs);
323  break;
324 
325  case SRC_OBJECT_TYPE + DOT + NO_PARENTHESES:
326  MClassifier srcClass = (MClassifier)srcType;
327  MAttribute attr = srcClass.attribute(opname, true);
328 
329  if (attr != null ) {
330  // (2) attribute operation on object type (no arguments)
331  res = new ExpAttrOp(attr, srcExpr);
332  } else {
333  // (4) navigation operation on object type
334  // must be a role name
335  MNavigableElement dst = srcClass.navigableEnd(opname);
336 
337  if (dst != null ){
338  res = genNavigation(fOp, srcClass, srcExpr, dst);
339  } else {
340  // (1) predefined OCL operation
341  res = genStdOperation(ctx, fOp, opname, fArgExprs);
342  }
343  }
344  break;
345 
346  case SRC_OBJECT_TYPE + DOT + NO_PARENTHESES + EXPLICIT_ROLENAME:
347  MClassifier srcClass3 = (MClassifier)srcType;
348  // (4) navigation operation on object type
349  // must be a role name
350  MNavigableElement dst = srcClass3.navigableEnd( opname );
351 
352  if (dst == null) {
353  throw new SemanticException(fOp, StringUtil.inQuotes(opname) + " is not a valid rolename that is reachable from class " + StringUtil.inQuotes(srcClass3.name()));
354  } else {
355  res = genNavigation( ctx, fOp, srcClass3, srcExpr, dst, fExplicitRolenameOrQualifiers, fQualifiers );
356  }
357 
358  break;
359 
360  case SRC_OBJECT_TYPE + DOT + PARENTHESES:
361  // (3) "isQuery" operation on object type (possibly with
362  // arguments) or (1)
363  MClassifier srcClass2 = (MClassifier)srcType;
364  res = genObjOperation(ctx, srcClass2, srcExpr);
365  break;
366 
367  case SRC_OBJECT_TYPE + ARROW + NO_PARENTHESES:
368  case SRC_OBJECT_TYPE + ARROW + PARENTHESES:
369  // (6) set operation on single object resulting from
370  // navigation over associations with multiplicity zero
371  // or one (p. 7-13 of OMG UML 1.3)
372  if (srcExpr instanceof ExpNavigation
373  || /* I'm to lazy to introduce an inheritance hierarchy for this */
374  srcExpr instanceof ExpNavigationClassifierSource) {
375  // first map object to set with internal operation
376  Expression mappedSrcExpr = new ExpObjAsSet(srcExpr);
377  // replace receiver in arg list
378  fArgExprs[0] = mappedSrcExpr;
379  try {
380  // lookup collection operation
381  res = ExpStdOp.create(opname, fArgExprs);
382  } catch (ExpInvalidException ex) {
383  throw new SemanticException(fOp, ex);
384  }
385  } else {
386  throw new SemanticException(fOp,
387  "An arrow operation treating a single object as a " +
388  "set may only be applied, if the object results " +
389  "from a navigation to an association end with " +
390  "multiplicity 0..1.");
391  }
392  break;
393 
394  case SRC_COLLECTION_TYPE + DOT + NO_PARENTHESES:
395  case SRC_COLLECTION_TYPE + DOT + NO_PARENTHESES + EXPLICIT_ROLENAME:
396  // c.op 200 (5) with implicit (2,4,1)
397  if (Options.disableCollectShorthand )
398  throw new SemanticException(fOp, MSG_DISABLE_COLLECT_SHORTHAND);
399  res = collectShorthandWithOutArgs(ctx, opname, srcExpr, fExplicitRolenameOrQualifiers, fQualifiers);
400  break;
401 
402  case SRC_COLLECTION_TYPE + DOT + PARENTHESES:
403  // c.op() 201 (5) with implicit (3,1)
404  if (Options.disableCollectShorthand )
405  throw new SemanticException(fOp, MSG_DISABLE_COLLECT_SHORTHAND);
406  res = collectShorthandWithArgs(ctx, opname, srcExpr);
407  break;
408 
409  case SRC_TUPLE_TYPE + DOT + PARENTHESES:
410  case SRC_TUPLE_TYPE + DOT + NO_PARENTHESES:
411  TupleType t = (TupleType)srcType;
412  TupleType.Part p = t.getPart(opname);
413  if (p==null) {
414  // try std operation
415  res = genStdOperation(ctx, fOp, opname, fArgExprs);
416  } else {
417  res = new ExpTupleSelectOp(p, srcExpr);
418  }
419  break;
420 
421  case SRC_TUPLE_TYPE + ARROW + PARENTHESES:
422  case SRC_TUPLE_TYPE + ARROW + NO_PARENTHESES:
423  throw new SemanticException(fOp, "Collection operation not applicable to tuple type.");
424 
425  default:
426  throw new RuntimeException("case " + Integer.toHexString(opcase) +
427  " not handled");
428  }
429 
430  if (isPre() )
431  res.setIsPre();
432  return res;
433  }

439  private Expression collectShorthandWithOutArgs(Context ctx, String opname, Expression srcExpr, List<ASTExpression> explicitRolenameOrQualifiers, List<ASTExpression> qualiferValues)
440  throws SemanticException
441  {
442  Expression res = null;
443  // c.op 200 (5) with implicit (2,4,1)
444  CollectionType cType = (CollectionType)srcExpr.type();
445  Type elemType = cType.elemType();
446 
447  if (elemType.isKindOfClassifier(VoidHandling.EXCLUDE_VOID)) {
448  MClassifier srcClass = (MClassifier)elemType;
449  MAttribute attr = srcClass.attribute(opname, true);
450 
451  if (attr != null ) {
452  // (2) attribute operation on object type (no arguments)
453 
454  // transform c.a into c->collect($e | $e.a)
455  ExpVariable eVar = new ExpVariable("$e", elemType);
456  ExpAttrOp eAttr = new ExpAttrOp(attr, eVar);
457  eAttr.setIsPre(this.isPre());
458  res = genImplicitCollect(srcExpr, eAttr, elemType);
459  } else {
460  MNavigableElement dst = srcClass.navigableEnd(opname);
461  if (dst != null ) {
462  // (4) navigation operation on object type must be
463  // a role name
464 
465  // transform c.r into c->collect($e | $e.r)
466  ExpVariable eVar = new ExpVariable("$e", elemType);
467  Expression eNav = genNavigation(ctx, fOp, srcClass, eVar, dst, explicitRolenameOrQualifiers, qualiferValues);
468  eNav.setIsPre(this.isPre());
469  res = genImplicitCollect(srcExpr, eNav, elemType);
470  }
471  }
472  } else if (elemType.isKindOfTupleType(VoidHandling.EXCLUDE_VOID)) {
473  TupleType t = (TupleType)elemType;
474  TupleType.Part p = t.getPart(opname);
475 
476  if (p != null) {
477  ExpVariable eVar = new ExpVariable("$e", elemType);
478  ExpTupleSelectOp eAttr = new ExpTupleSelectOp(p, eVar);
479  eAttr.setIsPre(this.isPre());
480  res = genImplicitCollect(srcExpr, eAttr, elemType);
481  }
482  }
483 
484  if (res == null ) {
485  // ! objectType || ! (attr || navigation)
486  // try (1) predefined OCL operation
487  res = collectShorthandStdOp(opname, srcExpr, cType, elemType);
488  }
489  return res;
490  }

497  private Expression collectShorthandWithArgs(Context ctx, String opname, Expression srcExpr)
498  throws SemanticException
499  {
500  Expression res = null;
501  // c.op() 201 (5) with implicit (3,1)
502  CollectionType cType = (CollectionType ) srcExpr.type();
503  Type elemType = cType.elemType();
504  if (elemType.isTypeOfClass() ) {
505  MClass srcClass = (MClass)elemType;
506  MOperation op = srcClass.operation(opname, true);
507  if (op != null ) {
508 
509  // operation must have a body
510  if (!op.hasBody()) {
511  throw new SemanticException(
512  fOp,
513  "Operation " +
514  srcClass.name() +
515  "::" +
516  opname +
517  " has no body.");
518  }
519 
520  // if the body is a soil statement, evaluating soil statements
521  // must be allowed
522  if (op.hasStatement()) {
523 
524  throw new SemanticException(
525  fOp,
526  "Operation " +
527  srcClass.name() +
528  "::" +
529  opname +
530  " is not a query operation.");
531  }
532 
533 
534  // transform c.op(...) into c->collect($e | $e.op(...))
535  ExpVariable eVar = new ExpVariable("$e", elemType);
536  fArgExprs[0] = eVar;
537  try {
538  // constructor performs additional checks
539  res = new ExpObjOp(op, fArgExprs);
540  res.setSourcePosition(new SrcPos(fOp));
541  } catch (ExpInvalidException ex) {
542  throw new SemanticException(fOp,
543  "In operation call `" + srcClass.name() + "::" +
544  opname + "': " + ex.getMessage());
545  }
546  res = genImplicitCollect(srcExpr, res, elemType);
547  }
548  }
549  if (res == null ) {
550  // try (1) predefined OCL operation
551  res = collectShorthandStdOp(opname, srcExpr, cType, elemType);
552  }
553  return res;
554  }

561  private Expression collectShorthandStdOp(String opname,
562  Expression srcExpr,
563  Type cType,
564  Type elemType)
565  throws SemanticException
566  {
567  Expression res = null;
568  // (1) predefined OCL operation
569 
570  // find operation on element type
571  Type[] params = new Type[fArgExprs.length];
572  params[0] = elemType;
573  for (int i = 1; i < fArgExprs.length; i++)
574  params[i] = fArgExprs[i].type();
575  if (! ExpStdOp.exists(opname, params) ) {
576  String msg = "Undefined operation `" + elemType +
577  "." + opname + "' in shorthand notation for collect.";
578  // maybe the user just confused . with ->, check for
579  // operation on collection and give a hint.
580  params[0] = cType;
581  if (ExpStdOp.exists(opname, params) )
582  msg += " However, there is an operation `" + cType +
583  "->" + opname +
584  "'. Maybe you wanted to use `->' instead of `.'?";
585  throw new SemanticException(fOp, msg);
586  }

588  fArgExprs[0] = new ExpVariable("$e", elemType);
589  try {
590  Expression eOp = ExpStdOp.create(opname, fArgExprs);
591  res = genImplicitCollect(srcExpr, eOp, elemType);
592  } catch (ExpInvalidException ex) {
593  // shouldn't fail because we already checked the
594  // existence above
595  throw new SemanticException(fOp, "collectShorthand failed: " + ex.getMessage());
596  }
597  return res;
