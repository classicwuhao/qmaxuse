56  private static <T> void combine(Stack<T> elementStack,
57  List<List<T>> listWithLists,
58  List<List<T>> combinations) {
59  if (elementStack.size() == listWithLists.size()) {
60  combinations.add( new ArrayList<T>(elementStack));
61  } else {
62  List<T> currentList = listWithLists.get(elementStack.size());
63 
64  for (T item : currentList) {
65  elementStack.push( item );
66  combine( elementStack, listWithLists, combinations );
67  elementStack.pop();
68  }
69  }
70  }

94  public static <T> List<List<Pair<T>>> combinationsOne( List<T> firstList, List<T> secondList, UniqueList unique ) {
95  List<T> l1 = (unique == UniqueList.FIRST_IS_UNIQUE ? secondList : firstList);
96  List<T> l2 = (unique == UniqueList.FIRST_IS_UNIQUE ? firstList : secondList);
97 
98  // Each partition contains only combinations from one element of the unique list
99  // e.g.: {a,b} {c,d} results in:
100  // 0 = {(a,c), (a,d)}, 1 = {(b,c), (b,d)}
101  List<List<Pair<T>>> partitionedCombinations = new ArrayList<List<Pair<T>>>();
102 
103  for (int index = 0; index < l1.size(); index++) {
104  List<Pair<T>> partition = new ArrayList<Pair<T>>();
105 
106  for (int index2 = 0; index2 < l2.size(); index2++) {
107  Pair<T> entry = new Pair<T>();
108  if (unique == UniqueList.FIRST_IS_UNIQUE) {
109  entry.first = l2.get(index2);
110  entry.second = l1.get(index);
111  } else {
112  entry.first = l1.get(index);
113  entry.second = l2.get(index2);
114  }
115  partition.add(entry);
116  }
117 
118  partitionedCombinations.add(partition);
119  }
120 
121 
122  List<List<Pair<T>>> result = new ArrayList<List<Pair<T>>>();
123  result.add(Collections.<Pair<T>>emptyList());
124  // Loops over all partitions of the combinations and calls combine.
125  // The tails is reduced by one after every call, because the previous call includes already the combinations
126  for (int index = 0; index < partitionedCombinations.size(); ++index) {
127  combinationsOneAux(result, Collections.<Pair<T>>emptyList(), partitionedCombinations.subList(index, partitionedCombinations.size()));
128  }
129 
130  return result;
131  }

133  private static <T> void combinationsOneAux(List<List<Pair<T>>> result, List<Pair<T>> head, List<List<Pair<T>>> tail) {
134 
135  if (tail.size() == 0) return;
136 
137  List<Pair<T>> myCombinations = tail.get(0);
138 
139  // All combination for a single element
140  // cmb = m * (1 + (|tail|
141  for(int index = 0; index < myCombinations.size(); ++index) {
142  List<Pair<T>> newHead = new LinkedList<Pair<T>>(head);
143  newHead.add(myCombinations.get(index));
144  result.add(newHead);
145  for (int index2 = 1; index2 < tail.size(); ++index2) {
146  combinationsOneAux(result, newHead, tail.subList(index2, tail.size()));
147  }
148  }
